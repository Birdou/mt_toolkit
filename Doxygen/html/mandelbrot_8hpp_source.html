21 void calc(Mt_pixelgrid &bitmap, const ld &minR, const ld &maxR, const ld &minI, const ld &maxI, int row, int column, size_t nElements)
22 {
23  size_t elements = 0;
24  for (size_t j = row; j < bitmap.height(); ++j)
25  {
26  for (size_t i = elements == 0 ? column : 0; i < bitmap.width(); ++i)
27  {
28  ld cr = i * ((maxR - minR) / bitmap.width()) + minR;
29  ld ci = j * ((maxI - minI) / bitmap.height()) + minI;
30 
31  int n = 0;
32  ld zr = 0.;
33  ld zi = 0.;
34  ld x2 = 0.;
35  ld y2 = 0.;
36  while (x2 + y2 <= 4. && n < maxN)
37  {
38  zi = 2. * zr * zi + ci;
39  zr = x2 - y2 + cr;
40  x2 = zr * zr;
41  y2 = zi * zi;
42  n++;
43  }
44 
45  // int value = (n * modifier) % 360;
46  // bitmap.at(i, j).hsv(value, 1, n == 360 ? 0 : 1);
47  bitmap.at(i, j).hex((n * modifier) % 16777216);
48 
49  elements++;
50  if (elements == nElements)
51  return;
52  }
53  }
54 }
55 void updateGrid(Mt_pixelgrid &bitmap, ld minR, ld maxR, ld minI)
56 {
57  std::vector<std::thread *> threads;
58  const size_t tElements = bitmap.width() * bitmap.height();
59  const size_t nElements = ceil(tElements / 4.);
60 
61  const ld maxI = minI + (maxR - minR) * bitmap.height() / bitmap.width();
62  for (size_t i = 0, j = 0; i < tElements; i += nElements, ++j)
63  {
64  threads.emplace_back(new std::thread(calc,
65  std::ref(bitmap),
66  std::ref(minR), std::ref(maxR),
67  std::ref(minI), std::ref(maxI),
68  i / bitmap.width(),
69  i % bitmap.width(),
70  nElements));
71  }
72  for (auto thread : threads)
73  {
74  thread->join();
75  delete thread;
76  }
77 }
